def raff_calculator(accepted_gaps, rejected_gaps):
    
    '''
    Returns the critical gap as estimated with Raff's method.

            Parameters:
                    accepted_gaps (numpy array or pandas series): data for the accepted gaps as floats
                    accepted_gaps (numpy array or pandas series): data for the rejected gaps as floats

            Returns:
                    gap (float): estimated critical gap, in the same units as the input data.
    '''
    
    # Sort data
    accepted = np.sort(accepted_gaps)
    rejected = np.sort(rejected_gaps)
    
    # Get cumulative probability of accepted gaps, and the complement of the cumulative probability of rejected gaps
    accepted_p = 1. * np.arange(len(accepted)) / (len(accepted) - 1)
    rejected_p = 1 - (1. * np.arange(len(rejected)) / (len(rejected) - 1))
    
    # Turn data into a line
    accepted_line = LineString(np.column_stack((accepted, accepted_p)))
    rejected_line = LineString(np.column_stack((rejected, rejected_p)))
    
    # Find the gap in the intersection of both curves according to Raff's definition
    gap = accepted_line.intersection(rejected_line).x
    return gap
